VAE Model¶

In [6]:
class VAE(nn.Module):
    def __init__(self, latent_dim):
        super(VAE, self).__init__()
        self.encoder = nn.Sequential(
            nn.Flatten(),
            nn.Linear(3 * image_size * image_size, 1024),
            nn.ReLU(),
            nn.Linear(1024, 512),
            nn.ReLU()
        )
        self.fc_mu = nn.Linear(512, latent_dim)
        self.fc_logvar = nn.Linear(512, latent_dim)

        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, 512),
            nn.ReLU(),
            nn.Linear(512, 1024),
            nn.ReLU(),
            nn.Linear(1024, 3 * image_size * image_size),
            nn.Tanh()
        )

    def reparameterize(self, mu, logvar):
        std = torch.exp(0.5 * logvar)
        eps = torch.randn_like(std)
        return mu + eps * std

    def forward(self, x):
        enc = self.encoder(x)
        mu = self.fc_mu(enc)
        logvar = self.fc_logvar(enc)
        z = self.reparameterize(mu, logvar)
        recon = self.decoder(z)
        return recon.view(-1, 3, image_size, image_size), mu, logvar

Loss Function¶

In [7]:
def loss_function(recon_x, x, mu, logvar):
    recon_loss = nn.functional.mse_loss(recon_x, x, reduction='sum')
    kld = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())
    return recon_loss + kld

def show_images(images, num_images=10):
    images = images[:num_images].cpu().detach()
    images = (images + 1) / 2  # rescale from [-1, 1] to [0, 1]
    grid = utils.make_grid(images, nrow=num_images, padding=2, normalize=False)
    plt.figure(figsize=(15, 2))
    plt.axis("off")
    plt.imshow(grid.permute(1, 2, 0).numpy())
    plt.show()

Training Loop¶

In [10]:
for epoch in range(1, num_epochs + 1):
    start_time = time.time()
    vae.train()
    total_loss = 0

    progress_bar = tqdm(dataloader, desc=f"Epoch {epoch}/{num_epochs}")
    for batch_idx, (data, _) in enumerate(progress_bar):
        data = data.to(device)
        optimizer.zero_grad()
        recon, mu, logvar = vae(data)
        loss = loss_function(recon, data, mu, logvar)
        loss.backward()
        optimizer.step()

        total_loss += loss.item()
        progress_bar.set_postfix(loss=loss.item())

    end_time = time.time()
    print(f"\nEpoch {epoch} finished in {end_time - start_time:.2f} seconds")
    print(f"Average Epoch Loss: {total_loss / len(dataloader):.2f}")
